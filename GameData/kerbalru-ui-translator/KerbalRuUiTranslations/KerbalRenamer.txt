# Единственная текстовая поверхность мода -- страница "Kerbal Renamer" в стоковом экране

# прохода KerbalRuUiTranslator.cs: страница рендерится через TMPro.TMP_Text, который тогда не
# патчился.
#
# Реальность подтверждена полным monodis-дизассемблированием Renamer.dll + Assembly-CSharp.dll:
# - RenamerCustomParams.get_Title()/.get_DisplaySection()/.get_Section() и custom-атрибуты
#   [GameParameters.CustomParameterUI("...")] / [GameParameters.CustomFloatParameterUI("...")] --
#   статичные литералы, не строятся через string.Concat/Format.
# - Прослежен полный путь до экрана: DifficultyOptionsMenu читает CustomParameterUI.get_title()
#   и создаёт new DialogGUIToggle(Func<bool>, string title, ...) / new DialogGUILabel(string, ...) /
#   new DialogGUIToggleButton(...) -- все три (через DialogGUIToggle.Create() и
#   DialogGUILabel.Create()) вызывают статический хелпер
#   DialogGUIBase.SetUpTextObject(TMPro.TextMeshProUGUI text, string value, ...), первая
#   инструкция которого -- `text.set_text(value)` БЕЗ каких-либо преобразований value.
#   Значит title-строка доходит до TMP_Text.text ровно в исходном виде -- патч ловит её.
# - Тултипы (toolTip) НЕ включены: они проходят через отдельное поле DialogGUIBase.tooltipText
#   и KSP.Localization.Localizer.Format(), путь до финального рендера не прослежен в этом
#   проходе -- не выдумываем перевод, который не проверен.
#
